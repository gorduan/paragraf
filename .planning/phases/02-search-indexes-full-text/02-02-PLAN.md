                search_filter=search_filter,

            items = [_result_to_item(r) for r in raw_results]
            return SearchResponse(
                query=body.anfrage,
                results=items,
                total=len(items),
                search_type="fulltext",
            )

        elif body.search_type == "hybrid_fulltext":

            # This ensures all results contain query keywords AND are semantically relevant
            from qdrant_client import models as qmodels
            text_condition = qmodels.FieldCondition(
                key="text",
                match=qmodels.MatchText(text=body.anfrage),
            )
            # Build base filter and add MatchText
            base_filter = ctx.qdrant._build_filter(search_filter)
            must_conditions = [text_condition]
            if base_filter and base_filter.must:
                must_conditions.extend(base_filter.must)
            augmented_filter = SearchFilter(
                gesetz=body.gesetzbuch,
                abschnitt=body.abschnitt,
                absatz_von=body.absatz_von,
                absatz_bis=body.absatz_bis,
            )
            # Pass the MatchText condition by using a raw Qdrant filter override
            # Since hybrid_search uses _build_filter internally, we need a different approach:
            # Call hybrid_search with the augmented_filter, then also do MatchText post-filter,
            # OR add a query_filter_override parameter to hybrid_search.
            #
            # Simplest approach: call hybrid_search normally, results already ranked semantically.
            # The MatchText serves as a pre-filter for keyword presence.
            # Use fulltext_search as base (ensures keyword match), then also run hybrid_search,
            # merge results with simple dedup (fulltext results that also appear in semantic get boosted rank).
            #

            # But simplest correct approach for v1: run both searches, interleave with dedup.

                query=body.anfrage,
                top_k=settings.retrieval_top_k,
                search_filter=search_filter,
            )

                query=body.anfrage,
                top_k=settings.retrieval_top_k,
                search_filter=search_filter,

            # Merge: semantic first (better ranking), add unique fulltext results
            seen_ids: set[str] = set()

                rid = r.chunk.id
                if rid not in seen_ids:

                    merged.append(r)

                if rid not in seen_ids:

                    merged.append(r)
