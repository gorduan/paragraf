```python
class ReferenceItem(BaseModel):
    gesetz: str
    paragraph: str
    absatz: int | None = None
    raw: str
    verified: bool
    kontext: str | None = None

class IncomingReferenceItem(BaseModel):
    gesetz: str = Field(description="Gesetz des zitierenden Paragraphen")
    paragraph: str = Field(description="Zitierender Paragraph")
    chunk_id: str = Field(description="Chunk-ID des zitierenden Punktes")
    text_preview: str = Field(description="Textvorschau (max 200 Zeichen)")

class ReferenceNetworkResponse(BaseModel):
    gesetz: str
    paragraph: str
    outgoing: list[ReferenceItem]
    incoming: list[IncomingReferenceItem]
    incoming_count: int

class DiscoverRequest(BaseModel):
    positive_ids: list[str] | None = None        # max 5
    positive_paragraphs: list[LookupRequest] | None = None  # max 5
    negative_ids: list[str] | None = None        # max 5
    negative_paragraphs: list[LookupRequest] | None = None   # max 5
    limit: int = 10                              # 1-50
    gesetzbuch: str | None = None
    abschnitt: str | None = None

class DiscoverResponse(BaseModel):
    positive_ids: list[str]
    negative_ids: list[str]
    results: list[SearchResultItem]
    total: int
    disclaimer: str
```
