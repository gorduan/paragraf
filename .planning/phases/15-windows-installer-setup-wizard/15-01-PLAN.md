```typescript
/** Persistenter App-Store fuer Setup-Wizard-State und Einstellungen. */
import Store from "electron-store";

export interface SetupState {
  setupComplete: boolean;
  setupStep: number;        // 0=welcome, 1=mode, 2=docker-check, 3=storage, 4=summary
  selectedMode: "docker";   // Only "docker" in v2.0
  dockerDetected: boolean;
  estimatedStorage: number; // MB
}

export interface AppSchema {
  setup: SetupState;
}

const defaults: AppSchema = {
  setup: {
    setupComplete: false,
    setupStep: 0,
    selectedMode: "docker",
    dockerDetected: false,
    estimatedStorage: 0,
  },
};

export const store = new Store<AppSchema>({ defaults });
```

```typescript
export interface DockerCheckResult {
  status: "running" | "installed" | "not-running" | "missing";
  version?: string;
}

export function checkDockerDetailed(): Promise<DockerCheckResult> {
  // Uses execFile (not exec) with array args for safety — no shell injection
  // Tier 1: docker info -> running
  // Tier 2: docker --version -> not-running
  // Tier 3: reg query -> installed
  // Tier 4: all fail -> missing
}
```

4. Add setup wizard IPC handlers to `desktop/src/main/ipc.ts`. Add imports for `shell` from electron, `store` from `./store`, `checkDockerDetailed` and `startDockerCompose` from `./docker`. Add 7 new handlers inside `registerIpcHandlers()` after existing docker handlers:

- `setup:getState` -> returns `store.get("setup")`
- `setup:setStep` -> receives step number, calls `store.set("setup.setupStep", step)`, returns true
- `setup:checkDocker` -> calls `checkDockerDetailed()`, updates `store.set("setup.dockerDetected", ...)` based on result status (true if running/not-running/installed), returns DockerCheckResult
- `setup:openDockerDownload` -> calls `shell.openExternal("https://www.docker.com/products/docker-desktop/")`, returns true
- `setup:complete` -> sets `store.set("setup.setupComplete", true)` and `store.set("setup.setupStep", 5)`, returns true
- `setup:storageEstimate` -> returns `{ dockerImages: 4000, mlModels: 4000, lawData: 500, total: 8500, unit: "MB" }`
- `setup:startDocker` -> logs "Setup abgeschlossen - Docker Compose starten", calls `startDockerCompose()`, returns `{ success: true }` on success or `{ success: false, error: message }` on failure. Uses try/catch with proper error extraction (`err instanceof Error ? err.message : String(err)`).

IMPORTANT: The `setup:startDocker` handler MUST be registered inside `registerIpcHandlers()` in `ipc.ts` alongside all other `setup:*` handlers. Do NOT put it in `desktop/src/main/index.ts`.
