```python
async def get_gpu_info() -> GpuInfoResponse:
    try:
        import torch
        if torch.cuda.is_available():
            name = torch.cuda.get_device_name(0)
            vram = int(torch.cuda.get_device_properties(0).total_memory / 1024 / 1024)
            return GpuInfoResponse(cuda_available=True, gpu_name=name, vram_total_mb=vram)
    except Exception:
        pass
    return GpuInfoResponse()
```

Methods:

- `__init__(self, hf_home: str | None = None)` — stores `self.hf_home = hf_home or os.environ.get("HF_HOME", "/models")`, creates `self._download_lock = asyncio.Lock()`, `self._downloading = False`

- `async def download_models(self) -> AsyncIterator[dict]` — per D-01, D-02, D-03, D-08, D-10:
  - Acquire `_download_lock`. If already locked, yield `{"type": "error", "message": "Download laeuft bereits"}` and return.
  - Set `self._downloading = True`.
  - Check disk space first (per D-10): `shutil.disk_usage(self.hf_home)`. Yield `{"type": "disk_check", "free_gb": round(disk.free / (1024**3), 1), "required_gb": 5.0}`. If free < 5GB, yield `{"type": "disk_warning", "free_gb": ..., "required_gb": 5.0}`.
  - For each model in MODELS:
    - Yield `{"type": "start", "model": model_id, "label": label}`
    - Use `huggingface_hub.list_repo_files(model_id)` to enumerate files. Filter to large files (`.safetensors`, `.bin`, `.model` > 1MB).
    - For each large file, call `huggingface_hub.hf_hub_download(model_id, filename, cache_dir=self.hf_home, tqdm_class=ProgressTqdm)` via `asyncio.to_thread()`. The custom `ProgressTqdm` (extending `tqdm.auto.tqdm`) overrides `update(self, n=1)` to call `super().update(n)` then invoke a callback with `(self.n, self.total)`.
    - **IMPORTANT per Pitfall 1 in RESEARCH.md:** Do NOT use `snapshot_download` for byte-level progress. Use `hf_hub_download()` per file with custom `tqdm_class` to get actual byte-level progress.
    - After each file, yield `{"type": "progress", "model": model_id, "downloaded_bytes": current, "total_bytes": total, "speed_mbps": speed, "eta_seconds": eta}`. Calculate speed from elapsed time and bytes downloaded.
    - After all files for a model, call `snapshot_download(model_id, cache_dir=self.hf_home)` in a thread to ensure all small config files are fetched too (this is fast since large files are already cached).
    - Yield `{"type": "complete", "model": model_id}`
    - On exception: implement retry logic (3 attempts with exponential backoff: 2s, 4s, 8s) per D-08. Yield `{"type": "retry", "model": model_id, "attempt": n, "max_attempts": 3}`. After 3 failures yield `{"type": "error", "model": model_id, "message": str(e), "retryable": True}`.
  - After all models: yield `{"type": "all_complete"}`.
  - Finally: set `self._downloading = False`, release lock.

- `async def get_model_status(self) -> dict` — returns `{"models_ready": bool, "models": [...], "downloading": self._downloading}`. For each model, check if `hub/models--{id}/` exists and contains `*.safetensors` files. Calculate size with `Path.rglob("*")`.

- `async def get_cache_info(self) -> dict` — returns `{"cache_path": self.hf_home, "total_size_mb": ..., "free_space_mb": ..., "models": [...]}`. Uses `shutil.disk_usage()` for free space. Walks `hub/` directory for per-model sizes.

- `async def clear_cache(self) -> dict` — removes `os.path.join(self.hf_home, "hub")` via `asyncio.to_thread(shutil.rmtree, hub_dir)`. Returns `{"erfolg": True, "nachricht": "Model-Cache geloescht"}`.

- `async def detect_gpu(self) -> dict` — per D-04 and Pitfall 2:
  - First try `nvidia-smi --query-gpu=name,memory.total --format=csv,noheader,nounits` via `asyncio.to_thread(subprocess.check_output, ...)` with 5s timeout.
  - Then try `torch.cuda.is_available()` and `torch.cuda.get_device_name(0)`.
  - Return `{"nvidia_smi_available": bool, "cuda_available": bool, "gpu_name": str, "vram_total_mb": int, "current_device": settings.embedding_device}`.

```python
class ModelDownloadEvent(BaseModel):
    """SSE-Event fuer Model-Download-Fortschritt."""
    type: Literal["disk_check", "disk_warning", "start", "progress", "complete", "retry", "error", "all_complete"]
    model: str | None = None
    label: str | None = None
    downloaded_bytes: int | None = None
    total_bytes: int | None = None
    speed_mbps: float | None = None
    eta_seconds: int | None = None
    free_gb: float | None = None
    required_gb: float | None = None
    message: str | None = None
    retryable: bool | None = None
    attempt: int | None = None
    max_attempts: int | None = None

class ModelInfo(BaseModel):
    """Info ueber ein einzelnes ML-Modell im Cache."""
    name: str
    label: str
    downloaded: bool = False
    size_mb: int = 0

class ModelStatusResponse(BaseModel):
    """Status aller ML-Modelle."""
    models_ready: bool = False
    models: list[ModelInfo] = Field(default_factory=list)
    downloading: bool = False

class ModelCacheEntry(BaseModel):
    """Ein Modell im Cache mit Groesse."""
    name: str
    size_mb: int = 0

class CacheInfoResponse(BaseModel):
    """Cache-Informationen fuer die Settings-Seite."""
    cache_path: str = ""
    total_size_mb: int = 0
    free_space_mb: int = 0
    models: list[ModelCacheEntry] = Field(default_factory=list)

class GpuDetectionResponse(BaseModel):
    """Erweiterte GPU-Erkennung mit nvidia-smi und torch.cuda."""
    nvidia_smi_available: bool = False
    cuda_available: bool = False
    gpu_name: str = ""
    vram_total_mb: int = 0
    current_device: str = "cpu"
```

```python
@app.post("/api/models/download")
async def download_models(request: Request) -> StreamingResponse:
    if model_manager._downloading:
        return JSONResponse(status_code=409, content={"error": "Download laeuft bereits"})

    async def _stream() -> AsyncIterator[str]:
        async for event in model_manager.download_models():
            if await request.is_disconnected():
                break
            yield f"data: {json.dumps(event)}\n\n"

    return StreamingResponse(_stream(), media_type="text/event-stream")
```
